SET(SOURCES ${SOURCES})
SET(HEADERS ${HEADERS})

SET(EDITOR_SOURCES ${EDITOR_SOURCES})
SET(EDITOR_HEADERS ${EDITOR_HEADERS})

SET(SUBDIRECTORIES
    platform
    rhi
    vulkan
)

FOREACH(DIRECTORY ${SUBDIRECTORIES})
    ADD_SUBDIRECTORY(${DIRECTORY})
ENDFOREACH()

SOURCE_GROUP(TREE ${CMAKE_CURRENT_SOURCE_DIR} PREFIX "sources" FILES ${SOURCES} ${HEADERS})

ADD_LIBRARY(
    engine
    SHARED
    ${SOURCES} ${HEADERS}
)

ADD_DEPENDENCIES(engine glfw)

TARGET_LINK_LIBRARIES(
    engine
    glfw
    ${VULKAN_LIBRARY}
    ${VULKAN_LAYER}
)
