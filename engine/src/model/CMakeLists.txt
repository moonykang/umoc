SET(SUBDIRECTORIES
    predefined
    gltf
)

FILE(GLOB ADDITIONAL_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB ADDITIONAL_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

FOREACH(DIRECTORY ${SUBDIRECTORIES})
    FILE(GLOB SUBDIRECTORY_ADDITIONAL_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/${DIRECTORY}/*.cpp)
    FILE(GLOB SUBDIRECTORY_ADDITIONAL_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/${DIRECTORY}/*.h)

    SET(ADDITIONAL_SOURCES ${ADDITIONAL_SOURCES} ${SUBDIRECTORY_ADDITIONAL_SOURCES})
    SET(ADDITIONAL_HEADERS ${ADDITIONAL_HEADERS} ${SUBDIRECTORY_ADDITIONAL_HEADERS})
ENDFOREACH()

SET(SOURCES ${SOURCES} ${ADDITIONAL_SOURCES} PARENT_SCOPE)
SET(HEADERS ${HEADERS} ${ADDITIONAL_HEADERS} PARENT_SCOPE)
