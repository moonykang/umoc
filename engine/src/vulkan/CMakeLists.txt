IF(PLATFORM_GLFW)
   SET(PLATFORM_PATH ${CMAKE_CURRENT_SOURCE_DIR}/platform/glfw)
ELSE()
   SET(PLATFORM_PATH ${CMAKE_CURRENT_SOURCE_DIR}/platform/android)
ENDIF()

SET(SUBDIRECTORIES
    resource
)

FOREACH(DIRECTORY ${SUBDIRECTORIES})
    FILE(GLOB ADDITIONAL_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/${SUBDIRECTORIES}/*.cpp)
    FILE(GLOB ADDITIONAL_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/${SUBDIRECTORIES}/*.h)
    SET(SOURCES ${SOURCES} ${ADDITIONAL_SOURCES} PARENT_SCOPE)
    SET(HEADERS ${HEADERS} ${ADDITIONAL_HEADERS} PARENT_SCOPE)
ENDFOREACH()

FILE(GLOB ADDITIONAL_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp ${PLATFORM_PATH}/*.cpp)
FILE(GLOB ADDITIONAL_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h ${PLATFORM_PATH}/*.h)

SET(SOURCES ${SOURCES} ${ADDITIONAL_SOURCES} PARENT_SCOPE)
SET(HEADERS ${HEADERS} ${ADDITIONAL_HEADERS} PARENT_SCOPE)
