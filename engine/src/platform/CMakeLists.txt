IF(PLATFORM_GLFW)
   SET(PLATFORM_PATH ${CMAKE_CURRENT_SOURCE_DIR}/glfw)

   SET(PLATFORM_PATH ${CMAKE_CURRENT_SOURCE_DIR}/android)
ENDIF()
